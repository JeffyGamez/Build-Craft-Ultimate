        function getWoodColors(woodType, blockType) {
            let logSide = '#746244', logTop = '#bfa16f', plank = '#bfa16f', leaves = '#2b5718', fence = '#9c7f51';
            if (woodType === 'spruce') {
                logSide = '#4b3820'; logTop = '#684e2c'; plank = '#684e2c'; leaves = '#1a390f'; fence = '#543f23';
            } else if (woodType === 'birch') {
                logSide = '#e5e7e9'; logTop = '#dfc090'; plank = '#dfc090'; leaves = '#446e29'; fence = '#caa977';
            } else if (woodType === 'acacia') {
                logSide = '#5d5750'; logTop = '#b05930'; plank = '#df6d38'; leaves = '#4c6119'; fence = '#b85425';
            }
            if (blockType === 'log') return { side: logSide, top: logTop, bottom: logTop };
            if (blockType === 'plank') return plank;
            if (blockType === 'leaves') return leaves;
            if (blockType === 'fence' || blockType === 'sapling') return fence;
        }

        function generateComplexTexture(config, side, drawingForInventoryIcon = false) {
            const canvas = document.createElement('canvas');
            canvas.width = 16; canvas.height = 16;
            const ctx = canvas.getContext('2d');
            let color = config.color || '#ffffff';

            if (config.type === 'grass') {
                if (side === 'top' || (drawingForInventoryIcon && side === 'all')) color = '#5b8731';
                else if (side === 'bottom') color = '#866043';
                else color = '#746244';
            }
            if (config.woodType) {
                const res = getWoodColors(config.woodType, config.type);
                if (config.type === 'log') {
                    if (side === 'top' || side === 'bottom' || (drawingForInventoryIcon && side === 'all')) color = res.top;
                    else color = res.side;
                } else {
                    color = res;
                }
            }
            
            if (config.type === 'glass') {
                ctx.clearRect(0,0,16,16);
                ctx.fillStyle = drawingForInventoryIcon ? 'rgba(150, 220, 255, 0.4)' : 'rgba(200, 240, 255, 0.15)';
                ctx.fillRect(0,0,16,16);
                ctx.strokeStyle = '#fff'; ctx.lineWidth = 1;
                ctx.beginPath(); ctx.moveTo(2,14); ctx.lineTo(14,2); ctx.stroke();
            } else if (config.type === 'tnt') {
                ctx.fillStyle = '#cc3333'; ctx.fillRect(0,0,16,16);
                ctx.fillStyle = '#ffffff'; ctx.fillRect(0,5,16,6);
                ctx.fillStyle = '#000000';
                ctx.fillRect(2,6,3,1); ctx.fillRect(3,7,1,3);
                ctx.fillRect(6,6,1,4); ctx.fillRect(7,7,1,1); ctx.fillRect(8,6,1,4);
                ctx.fillRect(11,6,3,1); ctx.fillRect(12,7,1,3);
                
                ctx.fillStyle = '#992222';
                for(let i=0; i<16; i+=2) { 
                    ctx.fillRect(i, 0, 1, 4); 
                    ctx.fillRect(i, 12, 1, 4); 
                }
            } else {
                ctx.fillStyle = color; ctx.fillRect(0, 0, 16, 16);
            }

            if (config.type !== 'glass' && config.type !== 'tnt') {
                ctx.fillStyle = 'rgba(0,0,0,0.05)';
                for(let i=0; i<12; i++) ctx.fillRect(Math.floor(Math.random()*16), Math.floor(Math.random()*16), 1, 1);

                if (config.type === 'bedrock') {
                    ctx.fillStyle = '#151618';
                    for(let i=0; i<16; i+=2) ctx.fillRect(0, i, 16, 1);
                    ctx.fillStyle = 'rgba(0,0,0,0.4)';
                    for(let i=0; i<20; i++) ctx.fillRect(Math.floor(Math.random()*16), Math.floor(Math.random()*16), 2, 1);
                } else if (config.type === 'sapling') {
                    ctx.fillStyle = '#2b5718'; ctx.fillRect(6,4,4,5);
                    ctx.fillStyle = '#746244'; ctx.fillRect(7,9,1,7);
                } else if (config.type === 'wool') {
                    ctx.fillStyle = 'rgba(0,0,0,0.08)';
                    for(let i=0; i<16; i+=2) { ctx.fillRect(0, i, 16, 1); ctx.fillRect(i, 0, 1, 16); }

        const materialPool = {};
        function getCachedMaterial(config, sideFacing = 'north', tntFlash = false) {
            const uid = `${config.id}_${sideFacing}_${tntFlash}`;
            if (materialPool[uid]) return materialPool[uid];

            if (tntFlash) {
                const whiteMat = new THREE.MeshStandardMaterial({ color: 0xffffff, roughness: 0.8 });
                materialPool[uid] = whiteMat; return whiteMat;
            }

            const isGlass = config.type === 'glass';
            const processTexture = (s) => {
                const tex = generateComplexTexture(config, s, false);
                tex.magFilter = THREE.NearestFilter; tex.minFilter = THREE.NearestFilter;
                return tex;
            };

            if (['grass', 'log', 'stove', 'chest', 'tnt'].includes(config.type)) {
                const s = 'side', t = 'top', b = 'bottom', f = 'front';
                let faces = [s, s, t, b, s, s];
                
                if (config.type === 'stove' || config.type === 'chest') {
                    if (sideFacing === 'north') faces = [s, s, t, b, f, s];
                    if (sideFacing === 'south') faces = [s, s, t, b, s, f];
                    if (sideFacing === 'east')  faces = [s, f, t, b, s, s];
                    if (sideFacing === 'west')  faces = [f, s, t, b, s, s];
                }

                const materials = faces.map(fName => new THREE.MeshStandardMaterial({ 
                    map: processTexture(fName), transparent: isGlass 
                }));
                materialPool[uid] = materials; return materials;
            } else {
                const singleTex = processTexture('all');
                const mat = new THREE.MeshStandardMaterial({ 
                    map: singleTex, transparent: isGlass || ['water','lava'].includes(config.type),
                    opacity: config.type === 'glass' ? 0.4 : (config.type==='water'?0.75:1.0)
                });
                materialPool[uid] = mat; return mat;
            }
        }

        function buildConnectedFenceMesh(x, y, z, data) {
            const group = new THREE.Group();
            const mat = getCachedMaterial(data.conf, 'north');
            
            const centerGeo = new THREE.BoxGeometry(0.26, 1.0, 0.26);
            const centerMesh = new THREE.Mesh(centerGeo, mat);
            group.add(centerMesh);

            const checkConnect = (dx, dy, dz) => {
                const neighbor = worldBlockData.get(`${x+dx},${y+dy},${z+dz}`);
                return neighbor && (neighbor.conf.type === 'fence' || neighbor.conf.type === 'log' || neighbor.conf.type === 'plank' || neighbor.conf.type === 'stone');
            };

            if (checkConnect(0, 0, -1)) {
                const m = new THREE.Mesh(new THREE.BoxGeometry(0.12, 0.15, 0.4), mat); m.position.set(0, 0.2, -0.3); group.add(m);
                const m2 = new THREE.Mesh(new THREE.BoxGeometry(0.12, 0.15, 0.4), mat); m2.position.set(0, -0.2, -0.3); group.add(m2);
            }
            if (checkConnect(0, 0, 1)) {
                const m = new THREE.Mesh(new THREE.BoxGeometry(0.12, 0.15, 0.4), mat); m.position.set(0, 0.2, 0.3); group.add(m);
                const m2 = new THREE.Mesh(new THREE.BoxGeometry(0.12, 0.15, 0.4), mat); m2.position.set(0, -0.2, 0.3); group.add(m2);
            }
            if (checkConnect(1, 0, 0)) {
                const m = new THREE.Mesh(new THREE.BoxGeometry(0.4, 0.15, 0.12), mat); m.position.set(0.3, 0.2, 0); group.add(m);
                const m2 = new THREE.Mesh(new THREE.BoxGeometry(0.4, 0.15, 0.12), mat); m2.position.set(0.3, -0.2, 0); group.add(m2);
            }
            if (checkConnect(-1, 0, 0)) {
                const m = new THREE.Mesh(new THREE.BoxGeometry(0.4, 0.15, 0.12), mat); m.position.set(-0.3, 0.2, 0); group.add(m);
                const m2 = new THREE.Mesh(new THREE.BoxGeometry(0.4, 0.15, 0.12), mat); m2.position.set(-0.3, -0.2, 0); group.add(m2);
            }

            group.position.set(x, y, z);
            return group;
        }

        function updateChunks() {
            const pX = Math.floor(camera.position.x);
            const pY = Math.floor(camera.position.y);
            const pZ = Math.floor(camera.position.z);

            for (const [key, mesh] of activeMeshes.entries()) {
                const [x, y, z] = key.split(',').map(Number);
                if (Math.abs(x - pX) > CHUNK_SIZE || Math.abs(y - pY) > CHUNK_SIZE || Math.abs(z - pZ) > CHUNK_SIZE) {
                    scene.remove(mesh); activeMeshes.delete(key);
                }
            }

            for (let x = pX - CHUNK_SIZE; x <= pX + CHUNK_SIZE; x++) {
                for (let y = pY - CHUNK_SIZE; y <= pY + CHUNK_SIZE; y++) {
                    for (let z = pZ - CHUNK_SIZE; z <= pZ + CHUNK_SIZE; z++) {
                        const key = `${x},${y},${z}`;
                        if (worldBlockData.has(key) && !activeMeshes.has(key)) {
                            const data = worldBlockData.get(key);
                            let mesh;
                            const isFlashing = activeTntAnimations.has(key) && activeTntAnimations.get(key).whiteMode;
                            
                            if (data.conf.type === 'fence') {
                                mesh = buildConnectedFenceMesh(x, y, z, data);
                            } else {
                                mesh = new THREE.Mesh(cubeGeometry, getCachedMaterial(data.conf, data.facing, isFlashing));
                                mesh.position.set(x, y, z);
                            }
                            scene.add(mesh); activeMeshes.set(key, mesh);
                        }
                    }
                }
            }
        }

                if(activeMeshes.has(k)) {
                    scene.remove(activeMeshes.get(k)); activeMeshes.delete(k);
                }
            });
        }

        function initJoystick(elementId, callback) {
            const el = document.getElementById(elementId);
            const stick = el.querySelector('.stick');
            let active = false, startX = 0, startY = 0;
            
            el.addEventListener('touchstart', (e) => {
                active = true; const t = e.touches[0]; const rect = el.getBoundingClientRect();
                startX = rect.left + rect.width/2; startY = rect.top + rect.height/2; e.preventDefault();
            }, { passive: false });
            
            window.addEventListener('touchmove', (e) => {
                if (!active) return;
                let touch = null;
                for(let t of e.touches) {
                    if(t.clientX > window.innerWidth/2 && elementId === 'right-joystick') touch = t;
                    if(t.clientX <= window.innerWidth/2 && elementId === 'left-joystick') touch = t;
                }
                if(!touch) touch = e.touches[0];
                let dx = touch.clientX - startX, dy = touch.clientY - startY;
                const radius = 35; const dist = Math.sqrt(dx*dx + dy*dy);
                if (dist > radius) { dx = (dx/dist)*radius; dy = (dy/dist)*radius; }
                stick.style.transform = `translate(${dx}px, ${dy}px)`;
                callback(dx / radius, dy / radius);
            });
            window.addEventListener('touchend', () => { if (active) { active = false; stick.style.transform = ''; callback(0,0); } });
        }

        function handleSaplingGrowth(x, y, z, woodType) {
            const logConf = blockTypes.find(b => b.type === 'log' && b.woodType === woodType);
            const leafConf = blockTypes.find(b => b.type === 'leaves' && b.woodType === woodType);
            
            for(let h = 0; h < 5; h++) {
                worldBlockData.set(`${x},${y+h},${z}`, { conf: logConf, facing: 'north' });
            }
            for(let ly = 2; ly <= 5; ly++) {
                let radius = ly === 5 ? 1 : 2;
                for(let lx = -radius; lx <= radius; lx++) {
                    for(let lz = -radius; lz <= radius; lz++) {
                        if(lx === 0 && lz === 0 && ly < 5) continue;
                        const lkey = `${x+lx},${y+ly},${z+lz}`;
                        if(!worldBlockData.has(lkey)) {
                            worldBlockData.set(lkey, { conf: leafConf, facing: 'north' });
                        }

                }
            }
        }

        function igniteAllTntInWorld() {
            for (const [key, val] of worldBlockData.entries()) {

        function triggerTntExplosion(centerX, centerY, centerZ) {
            igniteAllTntInWorld();

            for(let dx = -2; dx <= 2; dx++) {
                for(let dy = -2; dy <= 2; dy++) {
                    for(let dz = -2; dz <= 2; dz++) {
                        const targetX = centerX + dx;
                        const targetY = centerY + dy;
                        const targetZ = centerZ + dz;
                        const distance = Math.sqrt(dx*dx + dy*dy + dz*dz);
                        
                        if (distance <= 2.5) {
                            const bKey = `${targetX},${targetY},${targetZ}`;
                            if (worldBlockData.has(bKey)) {
                                if (worldBlockData.get(bKey).conf.type === 'bedrock') continue;
                                if (worldBlockData.get(bKey).conf.type === 'tnt') continue;

                                const mesh = activeMeshes.get(bKey);
                                if (mesh) scene.remove(mesh);
                                worldBlockData.delete(bKey); activeMeshes.delete(bKey);
                                rebuildFencesAround(targetX, targetY, targetZ);
                            }
                        }
                    }
                }
            }
            updateChunks();
        }

        function triggerSquareAction() {
            const currentSelection = blockTypes[currentSelectionIndex];
            actionTriggered = true;
            actionTime = 0;

            if (currentSelection.id === 'pickaxe') {
                isMiningAction = true;
                miningTime = 0;
                performMegaMine();
                return;
            }

            if (currentSelection.id === 'flint_steel') {
                raycaster.setFromCamera(screenCenter, camera);
                const hits = raycaster.intersectObjects(Array.from(activeMeshes.values()), true);
                if (hits.length > 0 && hits[0].distance < 6.5) {

                    while (targetedObj.parent && targetedObj.parent !== scene) { targetedObj = targetedObj.parent; }

                    if (worldBlockData.has(blockKey) && worldBlockData.get(blockKey).conf.type === 'tnt') {

                    }
                }
            }
        }

        function performBlockPlacement() {
            const currentSelection = blockTypes[currentSelectionIndex];
            if (currentSelection.type === 'item_tool' || ['sword', 'pickaxe', 'flint_steel'].includes(currentSelection.id)) {
                return; 
            }
            
            raycaster.setFromCamera(screenCenter, camera);
            const hits = raycaster.intersectObjects(Array.from(activeMeshes.values()), true);
            if (hits.length > 0 && hits[0].distance < 6.5) {
                let targetedObj = hits[0].object;
                while (targetedObj.parent && targetedObj.parent !== scene) { targetedObj = targetedObj.parent; }
                const targetPos = new THREE.Vector3().copy(targetedObj.position).add(hits[0].face.normal);
                const key = `${targetPos.x},${targetPos.y},${targetPos.z}`;
                
                if (!worldBlockData.has(key)) {
                    let facing = 'north';
                    const rad = yaw % (Math.PI * 2);
                    const angle = rad < 0 ? rad + Math.PI * 2 : rad;
                    if (angle >= 7 * Math.PI / 4 || angle < Math.PI / 4) facing = 'south';
                    else if (angle >= Math.PI / 4 && angle < 3 * Math.PI / 4) facing = 'west';
                    else if (angle >= 3 * Math.PI / 4 && angle < 5 * Math.PI / 4) facing = 'north';
                    else facing = 'east';

                    if (currentSelection.type === 'sapling') {
                        handleSaplingGrowth(targetPos.x, targetPos.y, targetPos.z, currentSelection.woodType);
                    } else {
                        worldBlockData.set(key, { conf: currentSelection, facing: facing });
                    }
                    rebuildFencesAround(targetPos.x, targetPos.y, targetPos.z);
                    updateChunks();
                }
            }
        }

        function performMegaMine() {
            raycaster.setFromCamera(screenCenter, camera);
            const hits = raycaster.intersectObjects(Array.from(activeMeshes.values()), true);
            if (hits.length > 0 && hits[0].distance < 6.5) {
                let targetedObj = hits[0].object;
                while (targetedObj.parent && targetedObj.parent !== scene) { targetedObj = targetedObj.parent; }
                const p = targetedObj.position;
                
                for (let dx = -1; dx <= 1; dx++) {
                    for (let dy = -1; dy <= 1; dy++) {
                        for (let dz = -1; dz <= 1; dz++) {
                            const tx = p.x + dx;
                            const ty = p.y + dy;
                            const tz = p.z + dz;
                            const targetKey = `${tx},${ty},${tz}`;
                            
                            if (worldBlockData.has(targetKey)) {
                                if (worldBlockData.get(targetKey).conf.type === 'bedrock') continue;
                                if (activeTntAnimations.has(targetKey)) activeTntAnimations.delete(targetKey);
                                const mesh = activeMeshes.get(targetKey);
                                if (mesh) scene.remove(mesh);
                                activeMeshes.delete(targetKey);
                                worldBlockData.delete(targetKey);
                                rebuildFencesAround(tx, ty, tz);
                            }
                        }
                    }
                }
                updateChunks();
            }
        }

        function performBlockDestruction() {
            raycaster.setFromCamera(screenCenter, camera);
            const hits = raycaster.intersectObjects(Array.from(activeMeshes.values()), true);
            if (hits.length > 0 && hits[0].distance < 6.5) {

                    rebuildFencesAround(p.x, p.y, p.z);
                    updateChunks();
                }
            }
        }

        function renderHandheldOverlay() {
            weaponCtx.clearRect(0, 0, weaponCanvas.width, weaponCanvas.height);
            const item = blockTypes[currentSelectionIndex];
            if (!item || !['sword', 'pickaxe', 'flint_steel'].includes(item.id)) return;

            weaponCtx.imageSmoothingEnabled = false;
            const cx = weaponCanvas.width / 2;
            const cy = weaponCanvas.height / 2;

            if (item.id === 'sword') {
                weaponCtx.save();
                if (isBlockingAction) {
                    weaponCtx.translate(weaponCanvas.width - 240, weaponCanvas.height - 180);
                    weaponCtx.rotate(-Math.PI / 4);
                    weaponCtx.scale(3.5, 3.5);
                } else if (actionTriggered) {
                    let progress = actionTime;
                    let angle = -Math.PI / 4 + Math.sin(progress * Math.PI) * -1.0;
                    weaponCtx.translate(weaponCanvas.width - 240 + Math.sin(progress * Math.PI) * -60, weaponCanvas.height - 200 + Math.cos(progress * Math.PI) * 30);
                    weaponCtx.rotate(angle);
                    weaponCtx.scale(3.8, 3.8);
                } else {
                    weaponCtx.translate(weaponCanvas.width - 240, weaponCanvas.height - 220);
                    weaponCtx.rotate(-Math.PI / 5);
                    weaponCtx.scale(3.8, 3.8);
                }

                // Shorter blade structure to remain well within view boundaries
                const swordMatrix = [
                    {dx:-3, dy:14, w:8, h:4, c:'#5c3a21'},
                    {dx:-3, dy:10, w:8, h:4, c:'#7a4f30'},
                    {dx:-3, dy:6,  w:8, h:4, c:'#5c3a21'},
                    {dx:-3, dy:2,  w:8, h:4, c:'#7a4f30'},
                    {dx:-13, dy:-3, w:28, h:5, c:'#2c3e50'},
                    {dx:-15, dy:-2, w:32, h:3, c:'#1f2d3d'},
                    {dx:-3, dy:-22, w:8, h:19, c:'#5ff2ff'},
                    {dx:-3, dy:-22, w:3, h:19, c:'#ffffff'}, 
                    {dx:2,  dy:-22, w:3, h:19, c:'#2b9fb3'}, 
                    {dx:-2, dy:-26, w:6, h:4, c:'#ffffff'},
                    {dx:-1, dy:-30, w:4, h:4, c:'#ffffff'}
                ];

                swordMatrix.forEach(box => {
                    weaponCtx.fillStyle = box.c;
                    weaponCtx.fillRect(box.dx * 4, box.dy * 4, box.w * 4, box.h * 4);
                });
                weaponCtx.restore();

            } else if (item.id === 'pickaxe') {
                weaponCtx.save();
                if (isMiningAction) {
                    let arc = Math.sin(miningTime * Math.PI) * -35;
                    weaponCtx.translate(weaponCanvas.width - 240, weaponCanvas.height - 200);
                    weaponCtx.rotate((-Math.PI / 4) + (arc * Math.PI / 180));
                } else {
                    weaponCtx.translate(weaponCanvas.width - 240, weaponCanvas.height - 200);
                    weaponCtx.rotate(-Math.PI / 4);
                }
                weaponCtx.scale(3.8, 3.8);

                // Shorter handle, with pickaxe head flairs systematically angled downward
                const pickMatrix = [
                    {dx:-3, dy:-2,  w:6, h:24, c:'#7a4f30'},
                    {dx:-3, dy:-2,  w:2, h:24, c:'#9c6f4c'}, 
                    {dx:1,  dy:-2,  w:2, h:24, c:'#5c3a21'}, 
                    
                    // Head center
                    {dx:-5, dy:-6, w:10, h:5, c:'#2c3e50'},
                    
                    // Left downward flair
                    {dx:-10, dy:-5, w:5, h:4, c:'#2b9fb3'},
                    {dx:-14, dy:-3, w:4, h:4, c:'#5ff2ff'},
                    {dx:-18, dy:0,  w:4, h:4, c:'#5ff2ff'},
                    {dx:-22, dy:4,  w:4, h:3, c:'#ffffff'}, 
                    
                    // Right downward flair
                    {dx:5,   dy:-5, w:5, h:4, c:'#2b9fb3'},
                    {dx:9,   dy:-3, w:4, h:4, c:'#5ff2ff'},
                    {dx:13,  dy:0,  w:4, h:4, c:'#5ff2ff'},
                    {dx:17,  dy:4,  w:4, h:3, c:'#2b9fb3'}
                ];

                pickMatrix.forEach(box => {
                    weaponCtx.fillStyle = box.c;
                    weaponCtx.fillRect(box.dx * 3.5, box.dy * 3.5, box.w * 3.5, box.h * 3.5);
                });
                weaponCtx.restore();

            } else if (item.id === 'flint_steel') {
                weaponCtx.save();
                if (actionTriggered) {
                    let stroke = Math.min(actionTime * 2.0, 1.0);
                    let tX = 60 + (cx - 100 - 60) * THREE.MathUtils.smoothstep(stroke, 0, 1);
                    let tY = (weaponCanvas.height - 180) + (cy - (weaponCanvas.height - 180)) * THREE.MathUtils.smoothstep(stroke, 0, 1);
                    weaponCtx.translate(tX, tY);
                } else {
                    weaponCtx.translate(60, weaponCanvas.height - 180);
                }
                weaponCtx.scale(4.5, 4.5);
                weaponCtx.fillStyle = '#2c3e50'; weaponCtx.fillRect(0,0,14,12);
                weaponCtx.fillStyle = '#1a252f'; weaponCtx.fillRect(2,2,8,6);
                weaponCtx.restore();

                weaponCtx.save();
                if (actionTriggered) {
                    let stroke = Math.min(actionTime * 2.0, 1.0);
                    let tX = (weaponCanvas.width - 140) + (cx + 20 - (weaponCanvas.width - 140)) * THREE.MathUtils.smoothstep(stroke, 0, 1);
                    let tY = (weaponCanvas.height - 200) + (cy - (weaponCanvas.height - 200)) * THREE.MathUtils.smoothstep(stroke, 0, 1);
                    weaponCtx.translate(tX, tY);
                } else {
                    weaponCtx.translate(weaponCanvas.width - 140, weaponCanvas.height - 200);
                }
                weaponCtx.scale(4.5, 4.5);
                weaponCtx.fillStyle = '#bdc3c7'; weaponCtx.fillRect(0,0,12,16);
                weaponCtx.fillStyle = '#7f8c8d'; weaponCtx.fillRect(0,0,4,16);
                weaponCtx.restore();
            }
        }

        function updateWeaponKeyframeAnimations() {
            if (actionTriggered) {
                actionTime += 0.03;
                if (actionTime >= 1.0) { actionTriggered = false; actionTime = 0; }
            }
            if (isMiningAction) {
                miningTime += 0.15;
                if (miningTime >= 1.0) { isMiningAction = false; miningTime = 0; }
            }
            renderHandheldOverlay();
        }

                iconCtx.imageSmoothingEnabled = false;

                if (['sword', 'pickaxe', 'flint_steel'].includes(block.id)) {
                    if (block.id === 'sword') {
                        iconCtx.fillStyle = '#5ff2ff'; iconCtx.fillRect(6,2,4,10); iconCtx.fillRect(4,12,8,2);
                    } else if (block.id === 'pickaxe') {
                        iconCtx.fillStyle = '#5ff2ff'; iconCtx.fillRect(3,2,10,3); iconCtx.fillRect(7,5,2,9);

        function isPositionColliding(x, y, z) {
            const checkPoints = [
                {x: x - 0.3, y: y - 1.4, z: z - 0.3}, {x: x + 0.3, y: y - 1.4, z: z - 0.3},
                {x: x - 0.3, y: y - 1.4, z: z + 0.3}, {x: x + 0.3, y: y - 1.4, z: z + 0.3},
                {x: x - 0.3, y: y, z: z - 0.3}, {x: x + 0.3, y: y, z: z - 0.3}
            ];
            for (let pt of checkPoints) {
                const k = `${Math.floor(pt.x + 0.5)},${Math.floor(pt.y + 0.5)},${Math.floor(pt.z + 0.5)}`;
                if (worldBlockData.has(k)) {
                    const type = worldBlockData.get(k).conf.type;
                    if (type !== 'water' && type !== 'lava') return true;
                }
            }
            return false;
        }

        const clock = new THREE.Clock();
        const moveSpeed = 6.0;

        updateChunks();

        function animate() {
            requestAnimationFrame(animate);

            const delta = Math.min(clock.getDelta(), 0.1);
            
            for (let [key, val] of activeTntAnimations.entries()) {
                val.clock += delta;
                let flashCycle = Math.floor(val.clock / 0.20) % 2 === 0;
                if (val.whiteMode !== flashCycle) {
                    val.whiteMode = flashCycle;
                    if (activeMeshes.has(key)) {
                        scene.remove(activeMeshes.get(key)); activeMeshes.delete(key);
                    }
                }
                if (val.clock >= 2.2) {
                    const [tx, ty, tz] = key.split(',').map(Number);
                    if (activeMeshes.has(key)) { scene.remove(activeMeshes.get(key)); activeMeshes.delete(key); }
                    worldBlockData.delete(key); activeTntAnimations.delete(key);
                    triggerTntExplosion(tx, ty, tz);
                }
            }

            if (currentMode === 'mobile') {
                yaw -= lookInput.x * 1.3 * delta; pitch += lookInput.y * 1.3 * delta;
            }
            pitch = Math.max(-Math.PI / 2.2, Math.min(Math.PI / 2.2, pitch));

            let combinedMoveX = 0, combinedMoveY = 0;
            if (currentMode === 'pc') {
                if (keysPressed['w'] || keysPressed['arrowup']) combinedMoveY += 1;
                if (keysPressed['s'] || keysPressed['arrowdown']) combinedMoveY -= 1;
                if (keysPressed['a'] || keysPressed['arrowleft']) combinedMoveX -= 1;
                if (keysPressed['d'] || keysPressed['arrowright']) combinedMoveX += 1;
            } else if (currentMode === 'mobile') {
                combinedMoveX = moveInput.x; combinedMoveY = moveInput.y;
            }

            const forward = new THREE.Vector3(0, 0, -1).applyAxisAngle(new THREE.Vector3(0, 1, 0), yaw);
            const right = new THREE.Vector3(1, 0, 0).applyAxisAngle(new THREE.Vector3(0, 1, 0), yaw);

            const newPos = camera.position.clone();
            
            newPos.addScaledVector(forward, combinedMoveY * moveSpeed * delta);
            newPos.addScaledVector(right, combinedMoveX * moveSpeed * delta);
            
            if (isFlyingMode) {
                if (flyUpActive || keysPressed[' ']) newPos.y += 4.5 * delta;
                if (flyDownActive || keysPressed['shift']) newPos.y -= 4.5 * delta;
            } else {
                verticalVelocity -= 9.8 * delta; 
                newPos.y += verticalVelocity * delta;

                if (isPositionColliding(camera.position.x, newPos.y, camera.position.z)) {
                    newPos.y = camera.position.y;
                    verticalVelocity = 0;
                    
                    if ((combinedMoveX !== 0 || combinedMoveY !== 0) && isPositionColliding(newPos.x, camera.position.y, newPos.z)) {
                        if (!isPositionColliding(newPos.x, camera.position.y + 1.1, newPos.z)) {
                            newPos.y += 1.0; 
                        }
                    }
                }
            }

            if (currentMode) {
                if (!isPositionColliding(newPos.x, newPos.y, camera.position.z)) camera.position.x = newPos.x;
                if (!isPositionColliding(camera.position.x, newPos.y, camera.position.z)) camera.position.y = newPos.y;
                if (!isPositionColliding(camera.position.x, newPos.y, newPos.z)) camera.position.z = newPos.z;
            }
            
            const edgeX = (worldWidth / 2) - 0.5; const edgeZ = (worldDepth / 2) - 0.5;
            if (camera.position.x > edgeX) camera.position.x = edgeX;
            if (camera.position.x < -edgeX) camera.position.x = -edgeX;
            if (camera.position.z > edgeZ) camera.position.z = edgeZ;
            if (camera.position.z < -edgeZ) camera.position.z = -edgeZ;

            camera.quaternion.setFromEuler(new THREE.Euler(pitch, yaw, 0, 'YXZ'));
            
            if (isBlockingAction) { camera.position.y -= 0.5; }

            updateWeaponKeyframeAnimations();
            updateChunks();
            renderer.render(scene, camera);

            if (isBlockingAction) { camera.position.y += 0.5; }
        }

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight; camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });
        animate();
    </script>
